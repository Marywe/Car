8bool process_events(Window& window, b2World& physics_world, float window_height, float scale, Car* car)
9{

30 car->Brake();
31 }
32

36 car->Reset();
37 }
38
39 break;
40 }
41 case Event::KeyReleased:
42 {
43 if (event.key.code == Keyboard::A || event.key.code == Keyboard::D) {
44 car->Brake();
45 }
46 break;
47 }
48
49 case Event::Closed:
50 {
51 return true;
52 }
53 }
54 }
55
56 return false;

273void render(b2World& physics_world, RenderWindow& renderer, float scale)
274{
275 // Se cachea el alto de la ventana en una variable local:
276
277 float window_height = (float)renderer.getSize().y;
278
279 // Se recorre toda la lista de bodies de physics_world:
280
281 for (b2Body* body = physics_world.GetBodyList(); body != nullptr; body = body->GetNext())
282 {
283 // Se obtiene el transform del body:
284
285 const b2Transform& body_transform = body->GetTransform();
286
287 // Se recorre la lista de fixtures del body:
288
289 for (b2Fixture* fixture = body->GetFixtureList(); fixture != nullptr; fixture = fixture->GetNext())
290 {
291 // Se obtiene el tipo de forma de la fixture:
292
293 b2Shape::Type shape_type = fixture->GetShape()->GetType();
294
295 switch (shape_type)
296 {
297 case b2Shape::e_circle:
298 {
299 // Se crea un CircleShape a partir de los atributos de la forma de la fixture y del body:
300 // En SFML el centro de un círculo no está en su position. Su position es la esquina superior izquierda
301 // del cuadrado en el que está inscrito. Por eso a position se le resta el radio tanto en X como en Y.
302
303 b2CircleShape* circle = dynamic_cast<b2CircleShape*>(fixture->GetShape());
304
305 render_circle(circle->m_p, circle->m_radius, body_transform, renderer, window_height, scale);
306
307 break;
308 }
309 case b2Shape::e_edge:
310 {
311 // Se toman los dos vértices del segmento y se ajusta su posición usando el transform del body.
312 // Los vértices resultantes se convierten y se ponen en un array para finalmente dibujar el segmento
313 // que los une usando la sobrecarga del método draw() que permite dibujar primitivas de OpenGL a
314 // partir de datos de vértices.
315
316 b2EdgeShape* edge = dynamic_cast<b2EdgeShape*>(fixture->GetShape());
317
318 render_segment(edge->m_vertex1, edge->m_vertex2, body_transform, renderer, window_height, scale);
319
320 break;
321 }
322
323 case b2Shape::e_polygon:
324 {
325 // Se toma la forma poligonal de Box2D (siempre es convexa) y se crea a partir de sus vértices un
326 // ConvexShape de SFML. Cada vértice de Box2D hay que transformarlo usando el transform del body.
327
328 b2PolygonShape* box2d_polygon = dynamic_cast<b2PolygonShape*>(fixture->GetShape());
329
330 render_polygon(box2d_polygon->m_vertices, box2d_polygon->m_count, body_transform, renderer, window_height, scale);
331
332 break;
333 }
334 }
335 }
336 }
337}
